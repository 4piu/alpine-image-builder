# Companion packages for this target's onboard AW859A wifi/BT -- see
# target/README.md and common/patches/, kernel.config, overlays/.
#
# Regulatory database -- CONFIG_CFG80211_REQUIRE_SIGNED_REGDB is on in
# the base kernel config, so cfg80211 refuses to enable any channels
# at all without this present in the rootfs.
wireless-regdb

# Userspace wifi stack -- confirmed on real hardware that wlan0 comes up
# (driver/firmware load succeeds, see setup.d/20-wifi-bt-modules.sh) but
# there's nothing here to actually scan/associate with: no iw, no
# wpa_supplicant. Both needed for anything beyond "the interface exists".
iw
wpa_supplicant
wpa_supplicant-openrc

# Userspace BT stack. The BT side of this chip talks HCI-over-UART
# via sprdbt_tty.ko's tty device, not a self-registering HCI device --
# something still needs to run hciattach against that tty at boot;
# modern bluez split hciattach out into bluez-deprecated. bluetoothd
# itself needs a running D-Bus system bus to register on -- confirmed on
# real hardware: bluetoothctl fails with a D-Bus connection assertion
# without it. Neither dbus nor bluetoothd start themselves just because
# the packages are installed -- see setup.d/40-enable-bt-services.sh for
# the rc-update wiring both need.
bluez
bluez-openrc
bluez-deprecated
dbus
dbus-openrc
